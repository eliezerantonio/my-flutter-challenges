import 'package:meta/meta.dart';
import 'dart:convert';
class PatientsResponse {
PatientsResponse({
@required this.street,
@required this.city,
@required this.state,
@required this.postcode,
@required this.coordinates,
});
String street;
String city;
String state;
String postcode;
Coordinates coordinates;
factory PatientsResponse.fromJson(String str) => PatientsResponse.fromMap(json.decode(str));
String toJson() => json.encode(toMap());
factory PatientsResponse.fromMap(Map<String, dynamic> json) => PatientsResponse(
street: json["street"],
city: json["city"],
state: json["state"],
postcode: json["postcode"],
coordinates: Coordinates.fromMap(json["coordinates"]),
);
Map<String, dynamic> toMap() => {
"street": street,
"city": city,
"state": state,
"postcode": postcode,
"coordinates": coordinates.toMap(),
};
}
